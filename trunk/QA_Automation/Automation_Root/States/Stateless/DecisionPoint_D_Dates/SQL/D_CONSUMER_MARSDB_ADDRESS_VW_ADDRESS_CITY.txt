SELECT AD.ADDRESS_CITY
         FROM MARSDB.MARSDB_ADDRESS_VW AD JOIN  MARSDB.MARSDB_CONTACTS_OWNER_VW CNO
          ON AD.CONTACT_TYPE_ID = CNO.CONTACT_OWNER_ID AND AD.PROJECT_ID = CNO.PROJECT_ID
WHERE     EXTERNAL_REF_ID = '31'
and       AD.PROJECT_ID = '109'
and       EXTERNAL_REF_TYPE = 'consumer';
