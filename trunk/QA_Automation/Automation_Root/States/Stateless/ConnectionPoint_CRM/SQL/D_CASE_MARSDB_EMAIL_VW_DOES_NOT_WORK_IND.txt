SELECT EML.DOES_NOT_WORK_IND
//       EML.PRIMARY_INDICATOR,
//       EML.DOES_NOT_WORK_IND,
//      CNO.EXTERNAL_REF_TYPE,
//       CNO.EXTERNAL_REF_ID,
//       CNO.PROJECT_ID
FROM MARS_DP4BI_DEV.MARSDB.MARSDB_EMAIL_VW EML 
JOIN  MARS_DP4BI_DEV.MARSDB.MARSDB_CONTACTS_OWNER_VW CNO
          ON EML.CONTACT_TYPE_ID = CNO.CONTACT_OWNER_ID 
          AND EML.PROJECT_ID = CNO.PROJECT_ID
WHERE EXTERNAL_REF_ID = 24 
and    CNO.PROJECT_ID = '44'
and    EXTERNAL_REF_TYPE = 'consumer';
