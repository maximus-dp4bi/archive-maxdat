         FROM MARS_DP4BI_DEV.MARSDB.MARSDB_PHONE_VW PHN JOIN  MARS_DP4BI_DEV.MARSDB.MARSDB_CONTACTS_OWNER_VW CNO
          ON PHN.CONTACT_TYPE_ID = CNO.CONTACT_OWNER_ID AND PHN.PROJECT_ID = CNO.PROJECT_ID
