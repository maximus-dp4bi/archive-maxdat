SELECT PHN.PHONE_NUMBER
         FROM MARSDB.MARSDB_PHONE_VW PHN
         JOIN  MARSDB.MARSDB_CONTACTS_OWNER_VW CNO
          ON PHN.CONTACT_TYPE_ID = CNO.CONTACT_OWNER_ID AND PHN.PROJECT_ID = CNO.PROJECT_ID
WHERE    EXTERNAL_REF_ID = '93'
and     PHN.PROJECT_ID = '44'
and       EXTERNAL_REF_TYPE = 'case';
