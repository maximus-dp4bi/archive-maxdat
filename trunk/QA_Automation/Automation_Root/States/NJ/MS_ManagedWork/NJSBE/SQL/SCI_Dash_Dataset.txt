with	 gopa1 as
 (select	a11.REPEAT_CONTACT_FLAG  YN_REPEATED_CONTACT,
		a11.CONTACT_SEQUENCE  CONTACT_SEQUENCE,
		a11.CONTACT_RECORD_ID  CONTACT_RECORD_ID,
		a11.PROJECT_ID  PROJECT_ID,
		a11.CONTACT_CREATING_TASKS_FLAG  CONTACT_CREATING_TASKS_FLAG,
		sum(a11.CONTACT_WRAP_UP_TIME)  WJXBFS1,
		sum(a11.CALL_DURATION)  WJXBFS2,
		count(distinct (a11.PROJECT_ID || a11.CONTACT_RECORD_ID))  WJXBFS3
	from	"PUBLIC"."D_CONTACT_RECORDS_SV"	a11
	group by	a11.REPEAT_CONTACT_FLAG,
		a11.CONTACT_SEQUENCE,
		a11.CONTACT_RECORD_ID,
		a11.PROJECT_ID,
		a11.CONTACT_CREATING_TASKS_FLAG
	)select	distinct a13.CONSUMER_CITY  CONSUMER_CITY,
	a13.CONSUMER_STATE  CONSUMER_STATE,
	a13.CONSUMER_AGE  CONSUMER_AGE,
	a13.CONSUMER_CORRESPONDENCE_PREFERENCE  CONSUMER_CORRESPONDENCE_PREFERENCE,
	a13.CONSUMER_COUNTY  CONSUMER_COUNTY,
	a13.CONSUMER_DOB  CONSUMER_DOB,
	a13.CONSUMER_GENDER  CONSUMER_GENDER,
	a14.CONSUMER_PREFERRED_LANGUAGE  CONSUMER_PREFERRED_LANGUAGE,
	a13.CONSUMER_ROLE  CONSUMER_ROLE,
	a13.CONSUMER_TYPE  CONSUMER_TYPE,
	a13.CONSUMER_ZIP_CODE  CONSUMER_ZIP_CODE,
	a13.CONSUMER_ZIP_CODE4  CONSUMER_ZIP_CODE_FOUR,
	a13.CONTACT_ACTION  CONTACT_ACTION,
	a13.CONTACT_CHANNEL  CONTACT_CHANNEL,
	a13.CONTACT_DISPOSITION  CONTACT_DISPOSITION,
	a13.CONTACT_REASON  CONTACT_REASON,
	pa12.CONTACT_RECORD_ID  CONTACT_RECORD_ID,
	pa12.PROJECT_ID  PROJECT_ID,
	a13.LINK_REFERENCE_ID  LINK_REFERENCE_ID,
	a13.CONTACT_RECORD_TYPE  CONTACT_RECORD_TYPE,
	a13.CONTACT_TYPE  CONTACT_TYPE,
	a13.CASE_ID  CASE_ID,
	a13.PROJECT_ID  PROJECT_ID0,
	a16.EXTERNAL_CASE_ID  EXTERNAL_CASE_ID,
	a13.CONSUMER_ID  CONSUMER_ID,
	a13.PROJECT_ID  PROJECT_ID1,
	a13.PROJECT_ID  PROJECT_ID2,
	a17.PROJECT_NAME  PROJECT_NAME0,
	a17.PROJECT_STATE  PROJECT_STATE,
	CAST(a13.CONTACT_CREATE_DATE_TIME AS DATE)  D_DATE,
	a15.D_MONTH  D_MONTH,
	pa12.CONTACT_SEQUENCE  CONTACT_SEQUENCE,
	pa12.YN_REPEATED_CONTACT  YN_REPEATED_CONTACT,
	pa12.CONTACT_CREATING_TASKS_FLAG  CONTACT_CREATING_TASKS_FLAG,
	pa12.WJXBFS1  WJXBFS1,
	pa12.WJXBFS3  WJXBFS2,
	pa12.WJXBFS2  WJXBFS3
from	gopa1	pa12
	left outer join	"PUBLIC"."D_CONTACT_RECORDS_SV"	a13
	  on 	(pa12.CONTACT_CREATING_TASKS_FLAG = a13.CONTACT_CREATING_TASKS_FLAG and 
	pa12.CONTACT_RECORD_ID = a13.CONTACT_RECORD_ID and 
	pa12.CONTACT_SEQUENCE = a13.CONTACT_SEQUENCE and 
	pa12.PROJECT_ID = a13.PROJECT_ID and 
	pa12.YN_REPEATED_CONTACT = a13.REPEAT_CONTACT_FLAG)
	left outer join	"PUBLIC"."D_CONSUMER_SV"	a14
	  on 	(a13.CONSUMER_ID = a14.CONSUMER_ID and 
	a13.PROJECT_ID = a14.PROJECT_ID)
	left outer join	"PUBLIC"."D_DATES"	a15
	  on 	(CAST(a13.CONTACT_CREATE_DATE_TIME AS DATE) = a15.D_DATE)
	left outer join	"PUBLIC"."D_CASE_SV"	a16
	  on 	(a13.CASE_ID = a16.CASE_ID and 
	a13.PROJECT_ID = a16.PROJECT_ID)
	left outer join	"PUBLIC"."D_PROGRAM_SV"	a17
	  on 	(a13.PROJECT_ID = a17.PROJECT_ID)
