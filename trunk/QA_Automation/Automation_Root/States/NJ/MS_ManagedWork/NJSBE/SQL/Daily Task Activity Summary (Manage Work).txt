select	a12.D_DATE  D_DATE,
	a11.TASK_TYPE_ID  TASK_TYPE_ID,
	a11.TASK_TYPE_NAME  TASK_NAME,
	a11.TASK_TYPE_DESCRIPTION  TASK_TYPE_DESCRIPTION,
	a11.OPERATIONS_GROUP  OPS_GROUP,
	a11.TASK_SLA_DAYS  SLA_DAYS,
	a11.TASK_SLA_JEOPARDY_DAYS  SLA_JEOPARDY_DAYS,
	a11.TASK_SLA_TARGET_DAYS  SLA_TARGET_DAYS,
	a11.TASK_SLA_TYPE  SLA_DAYS_TYPE,
	a11.TASK_PRIORITY  TASK_PRIORITY,
	a11.TASK_SOURCE  TASK_SOURCE,
	a11.TASK_STATUS  CURR_TASK_STATUS,
	a11.TASK_DISPOSITION  TASK_DISPOSITION,
	a11.TASK_CLAIM_STATUS  TASK_CLAIM_STATUS,
	a13.PROJECT_NAME  PROJECT_NAME,
	a11.PROJECT_ID  PROJECT_ID,
	sum(a11.CREATION_COUNT)  WJXBFS1,
	sum(a11.COMPLETION_COUNT)  WJXBFS2,
	sum(a11.CANCELLATION_COUNT)  WJXBFS3,
	sum(a11.TERMINATION_COUNT)  WJXBFS4,
	sum(a11.INVENTORY_COUNT)  INVENTORY_COUNT
from	"PUBLIC"."MW_F_TASK_INSTANCE_BY_DAY_SV"	a11
	left outer join	"PUBLIC"."D_DATES"	a12
	  on 	(a11.D_DATE = a12.D_DATE and 
	a11.PROJECT_ID = a12.PROJECT_ID)
	left outer join	"PUBLIC"."D_PROGRAM_SV"	a13
	  on 	(a11.PROJECT_ID = a13.PROJECT_ID)
group by	a12.D_DATE,
	a11.TASK_TYPE_ID,
	a11.TASK_TYPE_NAME,
	a11.TASK_TYPE_DESCRIPTION,
	a11.OPERATIONS_GROUP,
	a11.TASK_SLA_DAYS,
	a11.TASK_SLA_JEOPARDY_DAYS,
	a11.TASK_SLA_TARGET_DAYS,
	a11.TASK_SLA_TYPE,
	a11.TASK_PRIORITY,
	a11.TASK_SOURCE,
	a11.TASK_STATUS,
	a11.TASK_DISPOSITION,
	a11.TASK_CLAIM_STATUS,
	a13.PROJECT_NAME,
	a11.PROJECT_ID;
